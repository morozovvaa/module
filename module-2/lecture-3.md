- üîç –ù–∞—Ö–æ–¥–∏—Ç—å —Å–∫—Ä—ã—Ç—ã–µ –∑–∞–∫–æ–Ω–æ–º–µ—Ä–Ω–æ—Å—Ç–∏

## 2. –û–ø–∏—Å–∞—Ç–µ–ª—å–Ω–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞

### –ë–∞–∑–æ–≤—ã–µ –º–µ—Ç—Ä–∏–∫–∏ –≤ Python

```python
import pandas as pd
import numpy as np

# –û—Ü–µ–Ω–∫–∏ —É—á–µ–Ω–∏–∫–∞
grades = [5, 4, 5, 3, 4, 5, 4, 5]

# –û—Å–Ω–æ–≤–Ω—ã–µ –º–µ—Ç—Ä–∏–∫–∏
print("–°—Ä–µ–¥–Ω–µ–µ:", np.mean(grades))
print("–ú–µ–¥–∏–∞–Ω–∞:", np.median(grades))
print("–ú–æ–¥–∞:", pd.Series(grades).mode()[0])
print("–ú–∏–Ω:", np.min(grades))
print("–ú–∞–∫—Å:", np.max(grades))
print("–°—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ:", np.std(grades))
print("–†–∞–∑–º–∞—Ö:", np.max(grades) - np.min(grades))
```

### –ö–≤–∞—Ä—Ç–∏–ª–∏ –∏ –ø–µ—Ä—Ü–µ–Ω—Ç–∏–ª–∏

```python
# –°–æ–∑–¥–∞–¥–∏–º –¥–∞–Ω–Ω—ã–µ - –±–∞–ª–ª—ã –∑–∞ –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω—É—é
scores = [45, 67, 89, 78, 92, 56, 73, 85, 90, 88, 76, 
          82, 79, 91, 68, 74, 87, 83, 77, 94]

# –ö–≤–∞—Ä—Ç–∏–ª–∏
q1 = np.percentile(scores, 25)  # 25% —É—á–µ–Ω–∏–∫–æ–≤ –Ω–∞–±—Ä–∞–ª–∏ –º–µ–Ω—å—à–µ —ç—Ç–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è
q2 = np.percentile(scores, 50)  # –ú–µ–¥–∏–∞–Ω–∞
q3 = np.percentile(scores, 75)  # 75% —É—á–µ–Ω–∏–∫–æ–≤ –Ω–∞–±—Ä–∞–ª–∏ –º–µ–Ω—å—à–µ —ç—Ç–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è

print(f"–ü–µ—Ä–≤—ã–π –∫–≤–∞—Ä—Ç–∏–ª—å (Q1): {q1}")
print(f"–ú–µ–¥–∏–∞–Ω–∞ (Q2): {q2}")
print(f"–¢—Ä–µ—Ç–∏–π –∫–≤–∞—Ä—Ç–∏–ª—å (Q3): {q3}")
print(f"–ò–Ω—Ç–µ—Ä–∫–≤–∞—Ä—Ç–∏–ª—å–Ω—ã–π —Ä–∞–∑–º–∞—Ö (IQR): {q3 - q1}")

# –ò–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ü–∏—è
print(f"\n25% —É—á–µ–Ω–∏–∫–æ–≤ –Ω–∞–±—Ä–∞–ª–∏ –º–µ–Ω—å—à–µ {q1} –±–∞–ª–ª–æ–≤")
print(f"50% —É—á–µ–Ω–∏–∫–æ–≤ –Ω–∞–±—Ä–∞–ª–∏ –º–µ–Ω—å—à–µ {q2} –±–∞–ª–ª–æ–≤")
print(f"75% —É—á–µ–Ω–∏–∫–æ–≤ –Ω–∞–±—Ä–∞–ª–∏ –º–µ–Ω—å—à–µ {q3} –±–∞–ª–ª–æ–≤")
```

### –ê–Ω–∞–ª–∏–∑ —É—Å–ø–µ–≤–∞–µ–º–æ—Å—Ç–∏ –∫–ª–∞—Å—Å–∞

```python
import pandas as pd
import numpy as np

# –°–æ–∑–¥–∞–¥–∏–º –∂—É—Ä–Ω–∞–ª –∫–ª–∞—Å—Å–∞
np.random.seed(42)
journal = pd.DataFrame({
    '–£—á–µ–Ω–∏–∫': [f'–£—á–µ–Ω–∏–∫_{i}' for i in range(1, 26)],
    '–ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞': np.random.randint(2, 6, 25),
    '–†—É—Å—Å–∫–∏–π': np.random.randint(2, 6, 25),
    '–§–∏–∑–∏–∫–∞': np.random.randint(2, 6, 25),
    '–ò—Å—Ç–æ—Ä–∏—è': np.random.randint(2, 6, 25)
})

# –†–∞—Å—á—ë—Ç —Å—Ä–µ–¥–Ω–µ–≥–æ –±–∞–ª–ª–∞
subjects = ['–ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞', '–†—É—Å—Å–∫–∏–π', '–§–∏–∑–∏–∫–∞', '–ò—Å—Ç–æ—Ä–∏—è']
journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'] = journal[subjects].mean(axis=1).round(2)

# –û–±—â–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –∫–ª–∞—Å—Å–∞
print("=== –°–¢–ê–¢–ò–°–¢–ò–ö–ê –ö–õ–ê–°–°–ê ===")
print(f"–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —É—á–µ–Ω–∏–∫–æ–≤: {len(journal)}")
print(f"\n–°—Ä–µ–¥–Ω–∏–π –±–∞–ª–ª –∫–ª–∞—Å—Å–∞: {journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].mean():.2f}")
print(f"–ú–µ–¥–∏–∞–Ω–∞: {journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].median():.2f}")
print(f"–°—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ: {journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].std():.2f}")
print(f"–ú–∏–Ω–∏–º—É–º: {journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].min():.2f}")
print(f"–ú–∞–∫—Å–∏–º—É–º: {journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].max():.2f}")

# –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –ø—Ä–µ–¥–º–µ—Ç–∞–º
print("\n=== –°–¢–ê–¢–ò–°–¢–ò–ö–ê –ü–û –ü–†–ï–î–ú–ï–¢–ê–ú ===")
for subject in subjects:
    print(f"\n{subject}:")
    print(f"  –°—Ä–µ–¥–Ω–µ–µ: {journal[subject].mean():.2f}")
    print(f"  –ú–µ–¥–∏–∞–Ω–∞: {journal[subject].median():.2f}")
    print(f"  –ú–æ–¥–∞: {journal[subject].mode()[0]}")
    print(f"  –°—Ç–¥. –æ—Ç–∫–ª: {journal[subject].std():.2f}")

# –†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ø–æ –∫–∞—Ç–µ–≥–æ—Ä–∏—è–º
print("\n=== –†–ê–°–ü–†–ï–î–ï–õ–ï–ù–ò–ï –£–ß–ï–ù–ò–ö–û–í ===")
def categorize(avg):
    if avg >= 4.5: return '–û—Ç–ª–∏—á–Ω–∏–∫–∏'
    elif avg >= 3.5: return '–•–æ—Ä–æ—à–∏—Å—Ç—ã'
    elif avg >= 2.5: return '–¢—Ä–æ–µ—á–Ω–∏–∫–∏'
    else: return '–ù–µ—É—Å–ø–µ–≤–∞—é—â–∏–µ'

journal['–ö–∞—Ç–µ–≥–æ—Ä–∏—è'] = journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].apply(categorize)
print(journal['–ö–∞—Ç–µ–≥–æ—Ä–∏—è'].value_counts())

# Percentile analysis
print("\n=== –ü–†–û–¶–ï–ù–¢–ò–õ–ò ===")
percentiles = [10, 25, 50, 75, 90]
for p in percentiles:
    value = np.percentile(journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'], p)
    print(f"{p}% —É—á–µ–Ω–∏–∫–æ–≤ –∏–º–µ—é—Ç —Å—Ä–µ–¥–Ω–∏–π –±–∞–ª–ª –Ω–∏–∂–µ {value:.2f}")
```

### Z-score - –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –æ—Ü–µ–Ω–æ–∫

Z-score –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –Ω–∞—Å–∫–æ–ª—å–∫–æ –∑–Ω–∞—á–µ–Ω–∏–µ –æ—Ç–ª–∏—á–∞–µ—Ç—Å—è –æ—Ç —Å—Ä–µ–¥–Ω–µ–≥–æ –≤ –µ–¥–∏–Ω–∏—Ü–∞—Ö —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–≥–æ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è.

```python
# –†–∞—Å—á—ë—Ç z-score
journal['Z_score'] = (journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'] - journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].mean()) / journal['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'].std()

# –ò–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ü–∏—è
print("\n=== –ê–ù–ê–õ–ò–ó –û–¢–ù–û–°–ò–¢–ï–õ–¨–ù–û–ô –£–°–ü–ï–í–ê–ï–ú–û–°–¢–ò ===")
print("\n–£—á–µ–Ω–∏–∫–∏ —Å z-score > 1 (–∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω–æ –≤—ã—à–µ —Å—Ä–µ–¥–Ω–µ–≥–æ):")
print(journal[journal['Z_score'] > 1][['–£—á–µ–Ω–∏–∫', '–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª', 'Z_score']])

print("\n–£—á–µ–Ω–∏–∫–∏ —Å z-score < -1 (–∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω–æ –Ω–∏–∂–µ —Å—Ä–µ–¥–Ω–µ–≥–æ):")
print(journal[journal['Z_score'] < -1][['–£—á–µ–Ω–∏–∫', '–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª', 'Z_score']])
```

## 3. –ö–æ—Ä—Ä–µ–ª—è—Ü–∏–æ–Ω–Ω—ã–π –∞–Ω–∞–ª–∏–∑

### –ß—Ç–æ —Ç–∞–∫–æ–µ –∫–æ—Ä—Ä–µ–ª—è—Ü–∏—è?

–ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç —Å–≤—è–∑—å –º–µ–∂–¥—É –¥–≤—É–º—è –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–º–∏.

**–ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∫–æ—Ä—Ä–µ–ª—è—Ü–∏–∏ (r):**
- r = 1: –∏–¥–µ–∞–ª—å–Ω–∞—è –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–∞—è —Å–≤—è–∑—å
- r = 0.7-0.9: —Å–∏–ª—å–Ω–∞—è –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–∞—è —Å–≤—è–∑—å
- r = 0.4-0.6: —É–º–µ—Ä–µ–Ω–Ω–∞—è —Å–≤—è–∑—å
- r = 0: –Ω–µ—Ç —Å–≤—è–∑–∏
- r < 0: –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è —Å–≤—è–∑—å

### –ê–Ω–∞–ª–∏–∑ —Å–≤—è–∑–∏ –º–µ–∂–¥—É –ø—Ä–µ–¥–º–µ—Ç–∞–º–∏

```python
# –ö–æ—Ä—Ä–µ–ª—è—Ü–∏–æ–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞
correlation_matrix = journal[subjects].corr()
print("\n=== –ö–û–†–†–ï–õ–Ø–¶–ò–Ø –ú–ï–ñ–î–£ –ü–†–ï–î–ú–ï–¢–ê–ú–ò ===")
print(correlation_matrix.round(2))

# –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
import matplotlib.pyplot as plt
import seaborn as sns

plt.figure(figsize=(10, 8))
sns.heatmap(correlation_matrix, annot=True, fmt='.2f', 
            cmap='coolwarm', center=0, 
            square=True, linewidths=1)
plt.title('–ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è –æ—Ü–µ–Ω–æ–∫ –ø–æ –ø—Ä–µ–¥–º–µ—Ç–∞–º', fontsize=14, fontweight='bold')
plt.tight_layout()
plt.show()
```

### –ü—Ä–∏–º–µ—Ä: –°–≤—è–∑—å –ø–æ—Å–µ—â–∞–µ–º–æ—Å—Ç–∏ –∏ —É—Å–ø–µ–≤–∞–µ–º–æ—Å—Ç–∏

```python
# –°–æ–∑–¥–∞–¥–∏–º –¥–∞–Ω–Ω—ã–µ —Å –ø–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å—é
np.random.seed(42)
data = pd.DataFrame({
    '–£—á–µ–Ω–∏–∫': [f'–£—á–µ–Ω–∏–∫_{i}' for i in range(1, 31)],
    '–ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å_%': np.random.randint(60, 100, 30),
    '–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª': np.random.uniform(3.0, 5.0, 30)
})

# –î–æ–±–∞–≤–∏–º —Ä–µ–∞–ª–∏—Å—Ç–∏—á–Ω—É—é —Å–≤—è–∑—å
data['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'] = (data['–ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å_%'] / 25 + 
                        np.random.normal(0, 0.3, 30)).clip(2, 5)

# –†–∞—Å—á—ë—Ç –∫–æ—Ä—Ä–µ–ª—è—Ü–∏–∏
correlation = data['–ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å_%'].corr(data['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'])
print(f"\n–ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è –ø–æ—Å–µ—â–∞–µ–º–æ—Å—Ç–∏ –∏ —É—Å–ø–µ–≤–∞–µ–º–æ—Å—Ç–∏: {correlation:.3f}")

# –ò–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ü–∏—è
if correlation > 0.7:
    print("–í—ã–≤–æ–¥: –°–∏–ª—å–Ω–∞—è –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–∞—è —Å–≤—è–∑—å. –ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å –∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω–æ –≤–ª–∏—è–µ—Ç –Ω–∞ —É—Å–ø–µ–≤–∞–µ–º–æ—Å—Ç—å.")
elif correlation > 0.4:
    print("–í—ã–≤–æ–¥: –£–º–µ—Ä–µ–Ω–Ω–∞—è —Å–≤—è–∑—å. –ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å –≤–ª–∏—è–µ—Ç –Ω–∞ —É—Å–ø–µ–≤–∞–µ–º–æ—Å—Ç—å.")
else:
    print("–í—ã–≤–æ–¥: –°–ª–∞–±–∞—è —Å–≤—è–∑—å.")

# –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
plt.figure(figsize=(10, 6))
plt.scatter(data['–ü–æ—Å–µ—â–∞–µ–º–æ—Å—Ç—å_%'], data['–°—Ä–µ–¥–Ω–∏–π_–±–∞–ª–ª'],
```

### –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –¥–≤—É—Ö –≥—Ä—É–ø–ø (A/B —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ)

```python
if p_value < 0.05:
    print("\n‚úÖ –†–∞–∑–ª–∏—á–∏–µ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏ –∑–Ω–∞—á–∏–º–æ (p < 0.05)")
```
